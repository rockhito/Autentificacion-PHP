-- Crear tabla empleados
CREATE TABLE empleados (
  id_empleado INT PRIMARY KEY AUTO_INCREMENT,
  nombre VARCHAR(50),
  ciudad VARCHAR(50),
  cargo VARCHAR(50)

CREATE TABLE horarios_trabajados (
  id_horario INT PRIMARY KEY AUTO_INCREMENT,
  id_empleado INT,
  fecha DATE,
